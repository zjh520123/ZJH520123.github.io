<script>

    (function(){

        "use strict";

 

        // ---------- 模型常量 (与附表2、3完全一致) ----------

        const REG_COST_MONTH = [30, 80, 130];

        const REG_DECLINE_BASE = [0.02, 0.04, 0.06];

        const INTENSITY_COST_PER_SESSION = [3, 5, 8];

        const INTENSITY_EXTRA_DECLINE = [0, 0.03, 0.06];

        const BASE_FREQ = 5;

        const FREQ_EXTRA_PER_SESSION = 0.01;

        const WEEKS = 24;

        const MONTHS = WEEKS / 4;   // 6

        const MAX_BUDGET = 2000;

 

        // 年龄允许最大强度 (年龄组索引1~5)

        const MAX_INTENSITY_BY_AGE = [3, 3, 2, 2, 1];  // 索引0对应年龄组1

 

        // 活动评分允许最大强度

        function maxIntensityByScore(score) {

            if (score < 40) return 1;

            if (score < 60) return 2;

            return 3;

        }

 

        // 允许的调理分级 (根据初始积分)

        function allowedRegLevels(initPhlegm) {

            if (initPhlegm <= 58) return [1];

            if (initPhlegm >= 59 && initPhlegm <= 61) return [1, 2];

            return [1, 2, 3];

        }

 

        // ---------- 生成所有可行方案 ----------

        function generateSchemes(initPhlegm, ageGroup, actScore) {

            const allowedR = allowedRegLevels(initPhlegm);

            const maxS = Math.min(MAX_INTENSITY_BY_AGE[ageGroup - 1], maxIntensityByScore(actScore));

            const schemes = [];

 

            for (let r of allowedR) {

                for (let s = 1; s <= maxS; s++) {

                    for (let f = 1; f <= 10; f++) {

                        const cost = MONTHS * REG_COST_MONTH[r-1] + WEEKS * f * INTENSITY_COST_PER_SESSION[s-1];

                        if (cost > MAX_BUDGET) continue;

 

                        const baseDec = REG_DECLINE_BASE[r-1];

                        const intensityBonus = INTENSITY_EXTRA_DECLINE[s-1];

                        const freqBonus = Math.max(0, f - BASE_FREQ) * FREQ_EXTRA_PER_SESSION;

                        const monthlyDec = baseDec + intensityBonus + freqBonus;

                        

                        // 指数衰减: final = initial * (1 - monthlyDec)^6

                        const finalPhlegm = initPhlegm * Math.pow(1 - monthlyDec, MONTHS);

                        const decline = initPhlegm - finalPhlegm;

                        const declinePct = (decline / initPhlegm) * 100;

 

                        schemes.push({

                            r, s, f, cost,

                            finalPhlegm, decline, declinePct,

                            monthlyDec

                        });

                    }

                }

            }

            return schemes;

        }

 

        // ---------- 效果最优: 预算内最小化 finalPhlegm ----------

        function selectMaxEffect(schemes) {

            const valid = schemes.filter(s => s.cost <= MAX_BUDGET);

            if (valid.length === 0) throw new Error('无方案满足预算约束');

            return valid.reduce((best, cur) => cur.finalPhlegm < best.finalPhlegm ? cur : best);

        }

 

        // ---------- 性价比最优: 非支配前沿 + 最小ICER (不强制阈值，求最优单位成本效果) ----------

        function selectBestValue(schemes) {

            // 按成本升序排序

            const sorted = [...schemes].sort((a, b) => a.cost - b.cost);

            const n = sorted.length;

            

            // 剔除非支配方案 (成本更高且效果不更优)

            const keep = new Array(n).fill(true);

            for (let i = 0; i < n; i++) {

                for (let j = 0; j < n; j++) {

                    if (i === j) continue;

                    if (sorted[j].cost <= sorted[i].cost &&

                        sorted[j].finalPhlegm <= sorted[i].finalPhlegm &&

                        (sorted[j].cost < sorted[i].cost || sorted[j].finalPhlegm < sorted[i].finalPhlegm)) {

                        keep[i] = false;

                        break;

                    }

                }

            }

            const nondominated = sorted.filter((_, idx) => keep[idx]);

            

            if (nondominated.length === 0) throw new Error('无非支配方案');

            

            // 计算ICER，选择最小ICER方案 (效果提升相对成本的比率)

            let bestScheme = nondominated[0];

            let bestICER = Infinity;

            

            for (let i = 1; i < nondominated.length; i++) {

                const prev = nondominated[i-1];

                const curr = nondominated[i];

                const deltaCost = curr.cost - prev.cost;

                const deltaEffect = prev.finalPhlegm - curr.finalPhlegm;  // 下降值为正

                if (deltaEffect <= 1e-9) continue;  // 效果未改善

                const icer = deltaCost / deltaEffect;

                if (icer < bestICER) {

                    bestICER = icer;

                    bestScheme = curr;

                }

            }

            return bestScheme;

        }

 

        // ---------- 绘制柱状图 (Chart.js) ----------

        let chartEffectInstance = null;

        let chartValueInstance = null;

 

        function renderCharts(initPhlegm, planEffect, planValue) {

            const ctxEffect = document.getElementById('chartEffect').getContext('2d');

            const ctxValue = document.getElementById('chartValue').getContext('2d');

            

            // 销毁旧图表

            if (chartEffectInstance) chartEffectInstance.destroy();

            if (chartValueInstance) chartValueInstance.destroy();

 

            const labels = ['干预前', '干预后'];

            const effectData = [initPhlegm, planEffect.finalPhlegm];

            const valueData = [initPhlegm, planValue.finalPhlegm];

 

            // 效果最优图 (蓝色调)

            chartEffectInstance = new Chart(ctxEffect, {

                type: 'bar',

                data: {

                    labels: labels,

                    datasets: [{

                        label: '痰湿积分',

                        backgroundColor: ['#c0504e', '#4f81bd'],

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: {

                        legend: { display: false },

                        tooltip: { bodyFont: { size: 16 } }

                    },

                    scales: {

                        y: {

                            beginAtZero: true,

                            max: Math.max(initPhlegm, planEffect.finalPhlegm) * 1.25,

                            grid: { color: '#dbe2ea' },

                            title: {

                                display: true,

                                text: '痰湿积分',

                            },

                        }

                    }

                }

            });

 

            // 性价比最优图 (橙色调)

            chartValueInstance = new Chart(ctxValue, {

                type: 'bar',

                        label: '痰湿积分',

                        backgroundColor: ['#c0504e', '#e8a75e'],

                        borderRadius: 8,

                        barPercentage: 0.65,

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: {

                        legend: { display: false },

                        tooltip: { bodyFont: { size: 16 } }

                    },

                    scales: {

                        y: {

                            beginAtZero: true,

                            max: Math.max(initPhlegm, planValue.finalPhlegm) * 1.25,

                            grid: { color: '#dbe2ea' },

                            title: {

                                display: true,

                                text: '痰湿积分',

                                font: { size: 18, weight: '500' }

                            },

                            ticks: { font: { size: 16 } }

                        },

                        x: {

                            ticks: { font: { size: 18, weight: '500' } }

                        }

                    }

                }

            });

 

            // 在柱顶显示数值 (使用插件或内置注解? 简单起见直接用文字放在detail里，图表内通过tooltip查看)

            // 为了更清晰，我们可以在detail里已经展示了具体数值。

        }

 

        // 更新文本详情

        function updateDetailText(planEffect, planValue, initPhlegm) {

            const effectDiv = document.getElementById('effectDetail');

            const valueDiv = document.getElementById('valueDetail');

            

            effectDiv.innerHTML = `

            `;

            

            valueDiv.innerHTML = `

                <span>�� 总成本:</span> ${planValue.cost.toFixed(0)} 元   

                <span>�� 最终积分:</span> ${planValue.finalPhlegm.toFixed(1)} (下降 ${planValue.decline.toFixed(1)} , ${planValue.declinePct.toFixed(1)}%)

            `;

        }

 

        // 验证输入并执行分析

        function analyze() {

            const ageGroupInput = document.getElementById('ageGroup');

            const actScoreInput = document.getElementById('actScore');

            const phlegmInput = document.getElementById('phlegmInit');

            const errorSpan = document.getElementById('errorMsg');

            

            // 获取值

            const ageGroup = parseInt(ageGroupInput.value, 10);

            const actScore = parseFloat(actScoreInput.value);

            const initPhlegm = parseFloat(phlegmInput.value);

            

            // 清空错误

            errorSpan.textContent = '';

            

            // 验证

            if (isNaN(ageGroup) || ageGroup < 1 || ageGroup > 5 || !Number.isInteger(ageGroup)) {

                errorSpan.textContent = '❌ 年龄组必须为1~5的整数';

                return;

            }

            if (isNaN(actScore) || actScore < 0 || actScore > 100) {

                errorSpan.textContent = '❌ 活动总分需在0~100之间';

                return;

            }

            if (isNaN(initPhlegm) || initPhlegm < 0 || initPhlegm > 100) {

                errorSpan.textContent = '❌ 痰湿积分需在0~100之间';

                return;

            }

            

            try {

                const schemes = generateSchemes(initPhlegm, ageGroup, actScore);

                if (schemes.length === 0) {

                    errorSpan.textContent = '⚠️ 无可行方案 (预算内无解)';

                    return;

                }

                

                const planEffect = selectMaxEffect(schemes);

                const planValue = selectBestValue(schemes);

                

                // 更新文本和图表

                updateDetailText(planEffect, planValue, initPhlegm);

                renderCharts(initPhlegm, planEffect, planValue);

                

            } catch (e) {

                errorSpan.textContent = `⚠️ 计算错误: ${e.message}`;

                console.error(e);

            }

        }

 

        // 绑定事件 & 初始化默认示例

        document.addEventListener('DOMContentLoaded', () => {

            const btn = document.getElementById('analyzeBtn');

            btn.addEventListener('click', analyze);

            

            // 页面加载后自动运行一次 (默认数据 2, 38, 64)

            setTimeout(() => {

                analyze();

            }, 50);

        });
